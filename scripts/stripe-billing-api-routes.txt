
  if (url.pathname === "/billing/webhook" && req.method === "POST") {
    await handleStripeWebhookRoute(req, res);
    return;
  }

  if (url.pathname === "/billing/start" && req.method === "GET") {
    const session = getSession(req);
    if (!session) {
      redirect(res, `${PUBLIC_BASE_URL}/auth/discord`);
      return;
    }

    const guildId = url.searchParams.get("guild_id") || url.searchParams.get("server_id");
    const plan = url.searchParams.get("plan") || "premium_monthly";
    if (!/^\d{15,25}$/.test(String(guildId || ""))) {
      redirect(res, `${FRONTEND_URL}?upgrade=premium`);
      return;
    }
    if (!(await sessionCanAccessGuild(session, guildId))) {
      redirect(res, `${FRONTEND_URL}?billing=forbidden`);
      return;
    }
    const checkout = await createStripeCheckoutForGuild(guildId, session, plan);
    redirect(res, checkout.url);
    return;
  }

  if (url.pathname === "/api/billing/config" && req.method === "GET") {
    const session = requireSession(req, res);
    if (!session) return;
    json(res, 200, {
      configured: billingConfigured(),
      monthlyPriceUsd: Number(process.env.STOREBOT_PREMIUM_RENEWAL_MONTHLY_PRICE_USD || process.env.STOREBOT_PREMIUM_MONTHLY_PRICE_USD || 3),
      introEnabled: Boolean(STRIPE_INTRO_PRICE_ID || STRIPE_INTRO_COUPON_ID),
      allowPromotionCodes: STRIPE_ALLOW_PROMOTION_CODES
    }, req);
    return;
  }

  const billingStatusMatch = url.pathname.match(/^\/api\/guilds\/(\d{15,25})\/billing\/status$/);
  if (billingStatusMatch && req.method === "GET") {
    const session = requireSession(req, res);
    if (!session) return;
    const guildId = billingStatusMatch[1];
    if (!(await sessionCanAccessGuild(session, guildId))) {
      json(res, 403, { error: "You do not have access to this server." }, req);
      return;
    }
    json(res, 200, await getBillingStatusForGuild(guildId), req);
    return;
  }

  const billingCheckoutMatch = url.pathname.match(/^\/api\/guilds\/(\d{15,25})\/billing\/checkout$/);
  if (billingCheckoutMatch && req.method === "POST") {
    const session = requireSession(req, res);
    if (!session) return;
    const guildId = billingCheckoutMatch[1];
    if (!(await sessionCanAccessGuild(session, guildId))) {
      json(res, 403, { error: "You do not have access to this server." }, req);
      return;
    }
    const body = await readRequestJson(req, 8_192);
    json(res, 200, await createStripeCheckoutForGuild(guildId, session, body.plan || "premium_monthly"), req);
    return;
  }

  const billingPortalMatch = url.pathname.match(/^\/api\/guilds\/(\d{15,25})\/billing\/portal$/);
  if (billingPortalMatch && req.method === "POST") {
    const session = requireSession(req, res);
    if (!session) return;
    const guildId = billingPortalMatch[1];
    if (!(await sessionCanAccessGuild(session, guildId))) {
      json(res, 403, { error: "You do not have access to this server." }, req);
      return;
    }
    json(res, 200, await createStripeBillingPortalForGuild(guildId), req);
    return;
  }
